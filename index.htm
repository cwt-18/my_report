📄 測試檔

`timescale 1ns/1ns

module DDS_tb;
    reg clk;
    reg reset_n;
    reg[31:0]Fword;
    reg[3:0]Pword;
    wire[7:0]DA_Data;
        
    DDS DDS(
        .clk(clk),
        .reset_n(reset_n),
        .Fword(Fword),
        .Pword(Pword),
        .DA_Data(DA_Data)
    );

    initial clk = 1;
    always #5 clk = ~clk;
    
    initial begin
        reset_n = 0;
        Fword = 268435456;
        Pword = 15;
        #201;
        reset_n = 1;
    end

    // 計算
    // Fclk = 1 / 10ns = 100MHz
    // Fout = Fword × Fclk / 2^32
    //      = 268435456 × 100000000 / 4294967296
    //      = 6250000 ≅ 6.25MHz
endmodule
